	// Get a list of all exams created by a teacher
	Get All Exams (https://web.njit.edu/~pk549/490/beta/examTbl.php)
		Send:
		{
			requestType: 'getExams',
			ucid: teacherId
		}

		Receive:
		{
			exams: {
				0: examId,
				1: examId,
				etc.
			}
		}
